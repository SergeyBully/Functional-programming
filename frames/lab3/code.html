(define (SecantMethod func start end step)
(let ((xk (- start (/ (* (func start) (- end start)) (- (func end) (func start))))))
(cond
((> step (abs(- (func xk) (func start))))
xk)
((> 0(* (func xk) (func start)))
(SecantMethod func start xk step))
(else
(SecantMethod func xk end step)))))
